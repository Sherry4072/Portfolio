build_model <- function(region_code) {
  sub_data <- subset(data, region == region_code)
  sub_data <- data.frame(
    age = sub_data$age,
    wage = sub_data$income_level,
    gender = factor(sub_data$gender, levels = c(0, 1), labels = c("女", "男")),
    religion = factor(sub_data$religion, levels = c(0, 1), labels = c("無", "有")),
    marriage = factor(sub_data$marriage, levels = c(1, 2, 3, 4, 5, 6),
                      labels = c("從未結婚", "同居", "已婚", "離婚", "分居", "喪偶")),
    education = factor(sub_data$education, levels = c(0, 1, 2),
                       labels = c("大學", "高中以下", "碩博")),
    happiness = factor(sub_data$happiness, ordered = TRUE)
  )
  
  model <- polr(happiness ~  age + wage + gender + religion + marriage + education,
                data = sub_data, Hess = TRUE, na.action = na.exclude)
  return(model)
}
